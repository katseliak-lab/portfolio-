// Збір фідбеку зі Slack у Google Sheets
function collectSlackFeedback() {
  const messages = fetchChannelHistory(CHANNEL_ID);

  const parsed = messages
    .filter(msg => !isSystemMessage(msg))
    .map(msg => ({
      text: msg.text,
      sentiment: detectSentiment(msg.text),
      ts: msg.ts
    }));

  const sheet = SpreadsheetApp
    .getActive()
    .getSheetByName('Feedback');

  const existingIds = getExistingIds(sheet);

  parsed
    .filter(item => !existingIds.has(item.ts))
    .forEach(item => appendRow(sheet, item, 'New'));
}
